481 struct k_t *
482 k_crt_task (void (*pTask) (void), char prio, int stkSize, char *pStk)
483 {
484  struct k_t *pT;
485 
486 
487  int i;
488  char *s;
489 
490  if ((k_running) || ((prio <= 0) || (DMY_PRIO < prio))
491  || (k_task <= nr_task)) {
492  goto badexit;
493  }
494 
495  if (pStk == NULL)
496  {
497  goto badexit;
498  }
499  pT = task_pool + nr_task; // lets take a task descriptor
500  pT->nr = nr_task;
501  nr_task++;
502 
503  pT->cnt2 = 0; // no time out running on you for the time being
504  pT->cnt3 = 0; // no time out semaphore
505 
506  pT->cnt1 = (int) (pStk); // ref to my stack
507 
508  // stack paint :-)
509  for (i = 0; i < stkSize; i++) // put hash code on stak to be used by k_unused_stak()
510  {
511  pStk[i] = STAK_HASH;
512  }
513 
514  s = pStk + stkSize - 1; // now we point on top of stak
515  *(s--) = 0x00; // 1 byte safety distance :-)
516 
517  // an interrupt do only push PC on stack by HW - can be 2 or 3 bytes
518  // depending of 368/.../1280/2560
519 #ifdef BACKSTOPPER
520  pT->pt = pTask;
521  *(s--) = lo8 (jumper); // so top now holds address of function
522  *(s--) = hi8 (jumper); // which is code body for task
523 #else
524  *(s--) = lo8 (pTask); // so top now holds address of function
525  *(s--) = hi8 (pTask); // which is code body for task
526 #endif
527  // NB NB 2560 use 3 byte for call/ret addresses the rest only 2
528 #if defined (__AVR_ATmega2560__) || defined(__AVR_ATmega2561__)
529  *(s--) = EIND; // best guess : 3 byte addresses !!! or just 0
530 #endif
531 
532  // r1 is the socalled zero value register
533  // see https://gcc.gnu.org/wiki/avr-gcc
534  // can tmp be non zero (multiplication etc)
535  *(s--) = 0x00; // r1
536  *(s--) = 0x00; // r0
537  *(s--) = 0x00; // sreg
538 
539  //1280 and 2560 need to save rampz reg just in case
540 #if defined (__AVR_ATmega2560__) || defined (__AVR_ATmega1280__) || defined (__AVR_ATmega1284P__) || defined(__AVR_ATmega2561__)
541  *(s--) = RAMPZ; // best guess 0x3b
542  // obsolete JDN *(s--) = EIND; // best guess
543 #endif
544 
545 #if defined (__AVR_ATmega2560__) || defined (__AVR_ATmega1280__) || defined(__AVR_ATmega2561__)
546  *(s--) = EIND; // best guess 0x3c
547 #endif
548 
549  for (i = 0; i < 30; i++) //r2-r31 = 30 regs
550  {
551  *(s--) = 0x00;
552  }
553 
554  pT->sp_lo = lo8 (s); // now we just need to save stakptr
555  pT->sp_hi = hi8 (s); // in thread descriptor
556 
557  // HW DEPENDENT PART - ENDE
558 
559  pT->prio = prio; // maxv for holding org prio for inheritance
560  pT->maxv = (int) prio;
561  prio_enQ (pAQ, pT); // and put task in active Q
562 
563  return (pT);
564 
565 badexit:
566  k_err_cnt++;
567  return (NULL);
568 }
